export function PrintModule({module: [Name, Features]}) {
	return (
		<div>
			<h2>{Name}</h2>
			{Features.map(feature => <PrintFeature feature={feature}></PrintFeature>)}
			<hr></hr>
		</div>
	)
}

export function PrintFeature({feature: [Name, Tasks]}) {
	let totalManDays = Tasks.map(([_, ManDays]) => ManDays).reduce((x, y) => x + y)
	return (
		<div>
			<h4>{Name} (Total-Man-Days {totalManDays})</h4>
			{Tasks.map(task => <PrintTask task={task}></PrintTask>)}
			<br></br>
		</div>
	)
}

export function PrintTask({task: [Name, ManDays]}) {
	return (
		<div>
			<li>[{ManDays}]{Name}</li>
		</div>
	)
}


export function PrintTimeEstimate({TimeEstimate: [Modules, Metadata]}) {
	return (
		<div>
			{Modules.map(module => <PrintModule module={module}></PrintModule>)}
		</div>
	);
}
