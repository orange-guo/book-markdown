# Snowflake id generation


import SnowflakeId from 'snowflake-id'

import {useState} from 'react';

export const snowflake = new SnowflakeId({
	mid: 42,
	offset: (2019 - 1970) * 31536000 * 1000
});

export function Example() {
	const [id, setId] = useState(snowflake.generate());
	const [message, setMessage] = useState("No snowflake id generated yet");
	return (
		<div>
			<label>{message}</label>
			<br></br>
			<button onClick={() => {
				setId(snowflake.generate())
				setMessage("Snowflake id: " + id)
			}}>Click here to generate a snowflake id
			</button>
		</div>
	)
}

<Example></Example>
